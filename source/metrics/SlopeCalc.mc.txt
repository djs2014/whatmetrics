class SmoothSlopeCalc {
    private var filteredAltitudeHistory = [] as Array<Float>;
    private var maxWindowSize = 8; 

    // We store the exact interval that was used to capture each point
    private var intervalHistory = [] as Array<Float>; 

    function doMathematicalCalculation(currentInterval as Float) as Double {
        var n = filteredAltitudeHistory.size();
        if (n < 2) { return 0.0d; }

        // Calculate the total Rise (Current Altitude minus the Oldest Altitude)
        var totalRise = filteredAltitudeHistory[n - 1] - filteredAltitudeHistory[0];

        // Calculate the total Run by summing up the intervals stored in history
        var totalRun = 0.0f;
        for (var i = 0; i < intervalHistory.size(); i++) {
            totalRun += intervalHistory[i];
        }

        if (totalRun <= 0.0f) { return 0.0d; }

        // Grade % = (Rise / Run) * 100
        var grade = (totalRise / totalRun) * 100.0f;

        return grade.toDouble();
    }
}
